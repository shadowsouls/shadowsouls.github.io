<template>
  <div>
    <input
      type="text"
      :value="message"
      @input="$emit('update:message', $event.target.value)"
    />
  </div>
</template>

<script lang="ts">
// FIXED 这是一个已经完成的input组件，父组件只需要v-model:message='[变量]'即可使用
// 该实现相当于对v-model进行自定义
import { defineComponent } from "vue";

export default defineComponent({
  setup() {
    return {};
  },
  props: {
    message: {
      type: String,
      default: "",
    },
  },
});
</script>
